00043 //
00044 // shared_map - a shared_ptr to a std::map with std::map API
00045 //
00046 // shared_map extends std::map to support reference-counted
00047 // sharing.  The shared_map constructs a std::map by default,
00048 // but can be assigned to some other shared_map in order to
00049 // share that.
00050 //
00051 // Assignment of a std::map results in a (deep) copy.
00052 // Assignment of a shared_map is an additional reference.
00053 //
00054 // In principle std::map and shared_map should be API
00055 // compatible, with the exception of copying vs referencing.
00056 //
00057 // Example:
00058 //
00059 //   shared_map<int,int> a;
00060 //   a[1] = 2;
00061 //   a[2] = 4;
00062 //   a[3] = 8;
00063 //
00064 //   shared_map<int,int> b;
00065 //   b = a;                /* a and b now share the same map */
00066 //
00067 
00068 template<typename K, typename V>
00069 class shared_map : public shared_ptr< ::std::map<K,V> >
00070 {
00071   public:
00072 
00073     typedef ::std::map<K,V>    map;
00074     typedef shared_ptr<map>    parent;
00075     typedef ::std::size_t      size_type;
00076 
00077     typedef typename ::std::map<K,V>::iterator    iterator;
00078     typedef typename ::std::map<K,V>::const_iterator    const_iterator;
00079 
00080     inline shared_map()  : parent(new map()) {}
00081     inline ~shared_map() {}
00082 
00083     // Expose std::map API
00084 
00085     inline size_type size() const                { RegalAssert(parent::get()); return parent::get()->size(); }
00086 
00087     inline       V &operator[](const K &k)       { RegalAssert(parent::get()); return parent::get()->operator[](k); }
00088     inline const V &operator[](const K &k) const { RegalAssert(parent::get()); return parent::get()->operator[](k); }
00089 
00090     inline size_type erase(const K &k)           { RegalAssert(parent::get()); return parent::get()->erase(k);      }
00091     inline void      erase(iterator p)           { RegalAssert(parent::get()); return parent::get()->erase(p);      }
00092 
00093     inline size_type count(const K &k) const     { RegalAssert(parent::get()); return parent::get()->count(k);      }
00094 
00095     inline iterator       find(const K &k)       { RegalAssert(parent::get()); return parent::get()->find(k);      }
00096     inline iterator       end()                  { RegalAssert(parent::get()); return parent::get()->end();      }
00097     inline const_iterator find(const K &k) const { RegalAssert(parent::get()); return parent::get()->find(k);      }
00098     inline const_iterator end() const            { RegalAssert(parent::get()); return parent::get()->end();      }
00099 };
