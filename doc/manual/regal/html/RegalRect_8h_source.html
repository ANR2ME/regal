00098   template <typename T> void glRect(RegalContext *ctx, T x1, T y1, T x2, T y2)
00099   {
00100     // it's an error to call glRect between glBegin/glEnd, so just return when that happens.
00101 
00102     if (ctx->depthBeginEnd)
00103       return;
00104 
00105     DispatchTableGL &dt = ctx->dispatcher.emulation;
00106 
00107     // incrementing context->depthBeginEnd here avoids an assert in log_glBegin and
00108     // also keeps the log indentation as it should be
00109 
00110     ctx->depthBeginEnd++;
00111 
00112     dt.call(&dt.glBegin)(GL_POLYGON);
00113 
00114       Emu::glVertex2(dt, x1, y1);
00115       Emu::glVertex2(dt, x2, y1);
00116       Emu::glVertex2(dt, x2, y2);
00117       Emu::glVertex2(dt, x1, y2);
00118 
00119     // undo the above "artificial" increment
00120 
00121     ctx->depthBeginEnd--;
00122 
00123     dt.call(&dt.glEnd)();
00124   }
