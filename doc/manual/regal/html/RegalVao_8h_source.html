00220         //GLint e = tbl.glGetError();
00221         tbl.glBindBuffer( GL_ARRAY_BUFFER, a.buffer );

00224         lastBuffer = a.buffer;
00225       }
00226 
00227       EnableDisableVertexAttribArray( ctx, a.enabled, i );
00228       if( a.pointer || a.buffer )
00229       {
00230         tbl.glVertexAttribPointer( i, a.size, a.type, a.normalized, a.stride, a.pointer );
00231       }
00232     }
00233     if( lastBuffer != vao.vertexBuffer )

00240   {

00242     {

00255         if( objects.count( name ) == 0 )
00256         {

00260         name++;
