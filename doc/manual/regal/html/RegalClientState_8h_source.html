00783     GLsizei computeEffectiveStride(GLsizei stride, GLint size, GLenum type)
00784     {
00785       if (stride != 0)
00786         return stride;
00787 
00788       // when stride == 0, compute effectiveStride based on size and type
00789 
00790       if (size == GL_BGRA)
00791         return 4;
00792 
00793       switch (type)
00794       {
00795         case GL_BYTE:
00796         case GL_UNSIGNED_BYTE:
00797           break;
00798         case GL_INT_2_10_10_10_REV:
00799         case GL_UNSIGNED_INT_2_10_10_10_REV:
00800           // size must be 4 for these. Just return that.
00801           break;
00802         case GL_SHORT:
00803         case GL_UNSIGNED_SHORT:
00804         case GL_HALF_FLOAT:
00805           size *= 2;
00806           break;
00807         case GL_INT:
00808         case GL_UNSIGNED_INT:
00809         case GL_FLOAT:
00810         case GL_FIXED:
00811           size *= 4;
00812           break;
00813         case GL_DOUBLE:
00814           size *= 8;
00815           break;
00816         default:
00817           size = 0;
00818           break;
00819       }
00820 
00821       return size;
00822     }

01087     void glInterleavedArrays(GLenum format, GLsizei stride, const GLvoid *pointer)
01088     {
01089       if (vertexArrayBinding)
01090         return;
01091 
01092       // from glspec43.compatibility.20130214.withchanges.pdf
01093       // p. 365, section 10.5.1 "Interleaved Arrays"
01094       //
01095       // set et; ec; en; st; sc; sv; tc; pc; pn; pv; and s as a function
01096       // of table 10.6 and the value of format.
01097 
01098       GLboolean et, ec, en;
01099       GLenum    tc;
01100       GLsizei   st, sc, sv;
01101       GLsizei   pc, pn, pv;
01102       GLsizei   s;
01103 
01104       const int f = sizeof(GL_FLOAT);
01105       const int c = 4 * sizeof(GL_UNSIGNED_BYTE);
01106 
01107       switch(format)
01108       {
01109         case GL_V2F:
01110           et = GL_FALSE;
01111           ec = GL_FALSE;
01112           en = GL_FALSE;
01113           tc = GL_FLOAT;
01114           st = 0;
01115           sc = 0;
01116           sv = 2;
01117           pc = 0;
01118           pn = 0;
01119           pv = 0;
01120           pv = 0;
01121           s  = 2 * f;
01122           break;
01123         case GL_V3F:
01124           et = GL_FALSE;
01125           ec = GL_FALSE;
01126           en = GL_FALSE;
01127           tc = GL_FLOAT;
01128           st = 0;
01129           sc = 0;
01130           sv = 3;
01131           pc = 0;
