00080   GLuint Gen()
00081   {
00082     Name name;
00083     gen( 1, & name.drv );
00084     // could be more clever here, and this could fail...
00085     name.app = name.drv;
00086     const GLuint searchLimit = 1000000000;
00087     for( GLuint i = 0; i < searchLimit; i++ ) {
00088         if( app2drv.count( name.app ) == 0 ) {
00089             break;
00090         }
00091         name.app++;
00092     }
00093     if( ( name.app - name.drv ) >= searchLimit ) {
00094         return 0;
00095     }
00096     app2drv[ name.app ] = name;
00097     drv2app[ name.drv ] = & app2drv[ name.app ];
00098     return name.app;
00099   }

00101   GLuint ToDriverName( GLuint appName )
00102   {
00103     if( app2drv.count( appName ) == 0 )
00104     {
00105         Name & name = app2drv[ appName ];
00106         name.app = appName;
00107         gen( 1, & name.drv );
00108         drv2app[ name.drv ] = &name;
00109     }
00110     return app2drv[ appName ].drv;
00111   }

00113   GLuint ToAppName( GLuint drvName )
00114   {
00115     if( drv2app.count( drvName ) )
00116     {
00117         RegalAssert( drv2app[ drvName ] != NULL );
00118         return drv2app[ drvName ]->app;
00119     }
00120     return 0;
00121   }

00123   void Delete( GLuint appName )
00124   {
00125     if( appName == 0 || app2drv.count( appName ) == 0 ) {
00126         return;
00127     }
00128     Name n = app2drv[ appName ];
00129     app2drv.erase( n.app );
00130     RegalAssert( drv2app.count( n.drv ) != 0 );
00131     drv2app.erase( n.drv );
00132     del( 1, & n.drv );
00133   }
