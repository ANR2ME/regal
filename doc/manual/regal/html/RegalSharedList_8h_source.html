00043 //
00044 // shared_list - a shared_ptr to a std::list with std::list API
00045 //
00046 // shared_list extends std::list to support reference-counted
00047 // sharing.  The shared_list constructs a std::list by default,
00048 // but can be assigned to some other shared_list in order to
00049 // share that.
00050 //
00051 // Assignment of a std::list results in a (deep) copy.
00052 // Assignment of a shared_list is an additional reference.
00053 //
00054 // In principle std::list and shared_list should be API
00055 // compatible, with the exception of copying vs referencing.
00056 //
00057 // Example:
00058 //
00059 //   shared_list<int> a;
00060 //   a.push_back(1);
00061 //   a.push_back(2);
00062 //   a.push_back(3);
00063 //
00064 //   shared_list<int> b;
00065 //   b = a;                /* a and b now share the same list */
00066 //
00067 
00068 template<typename T>
00069 class shared_list : public shared_ptr< ::std::list<T> >
00070 {
00071   public:
00072 
00073     typedef          ::std::list<T>       list;
00074     typedef          shared_ptr<list>     parent;
00075     typedef          ::std::size_t        size_type;
00076     typedef typename list::iterator       iterator;
00077     typedef typename list::const_iterator const_iterator;
00078 
00079     inline shared_list()  : parent(new list()) {}
00080     inline ~shared_list() {}
00081 
00082     // Expose std::list API
00083 
00084     inline size_type size()        const { RegalAssert(parent::get()); return parent::get()->size();  }
00085 
00086     inline void push_front(const T &x)   { RegalAssert(parent::get()); parent::get()->push_front(x);  }
00087     inline void push_back (const T &x)   { RegalAssert(parent::get()); parent::get()->push_back(x);   }
00088 
00089     inline       iterator begin ()       { RegalAssert(parent::get()); return parent::get()->begin(); }
00090     inline const_iterator begin () const { RegalAssert(parent::get()); return parent::get()->begin(); }
00091 
00092     inline       iterator end ()         { RegalAssert(parent::get()); return parent::get()->end();   }
00093     inline const_iterator end ()   const { RegalAssert(parent::get()); return parent::get()->end();   }
00094 
00095     inline void remove(const T &x)       { RegalAssert(parent::get()); parent::get()->remove(x);      }
00096 };
